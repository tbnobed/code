### 2. Update server.js
Replace your current `server.js` with this complete implementation:

```javascript
const express = require('express');
const path = require('path');
const basicAuth = require('basic-auth');
const sqlite3 = require('sqlite3').verbose();
const app = express();
const PORT = process.env.PORT || 3000;

// Initialize database
const db = new sqlite3.Database('./appointments.db');
db.serialize(() => {
  db.run(`CREATE TABLE IF NOT EXISTS appointments (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    customer TEXT,
    date TEXT,
    time TEXT,
    service TEXT,
    phone TEXT,
    email TEXT
  )`);
});

// Admin middleware - requires basic auth
const authMiddleware = (req, res, next) => {
  const credentials = basicAuth(req);
  if (!credentials || credentials.name !== 'admin' || credentials.pass !== 'securepassword123') {
    res.setHeader('WWW-Authenticate', 'Basic realm="Admin Area"');
    return res.status(401).send('Unauthorized');
  }
  next();
};

// Admin routes
app.use('/admin', authMiddleware);

// Calendar page
app.get('/admin/calendar', (req, res) => {
  res.sendFile(path.join(__dirname, 'admin.html'));
});

// API for appointments data
app.get('/api/appointments', authMiddleware, (req, res) => {
  db.all('SELECT * FROM appointments', (err, rows) => {
    if (err) {
      res.status(500).send('Error fetching appointments');
      return;
    }
    res.json(rows);
  });
});

// Add new appointment
app.post('/api/appointments', authMiddleware, (req, res) => {
  const { customer, date, time, service, phone, email } = req.body;
  db.run(`INSERT INTO appointments (customer, date, time, service, phone, email) VALUES (?, ?, ?, ?, ?, ?)`,
    [customer, date, time, service, phone, email], (err) => {
      if (err) {
        res.status(500).send('Error adding appointment');
        return;
      }
      res.status(201).json({ success: true });
    });
});

// Delete appointment
app.delete('/api/appointments/:id', authMiddleware, (req, res) => {
  const id = req.params.id;
  db.run(`DELETE FROM appointments WHERE id = ?`, [id], (err) => {
    if (err) {
      res.status(500).send('Error deleting appointment');
      return;
    }
    res.status(200).json({ success: true });
  });
});

// Serve static files for main site
app.use(express.static(path.join(__dirname)));

app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
```
